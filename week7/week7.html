PROMISES

const dice = {
sides: 6,
roll() {
return Math.floor(this.sides * Math.random()) + 1;
}
}
const promise = new Promise( (resolve,reject) => {
const n = dice.roll();
setTimeout(() => {
(n > 1) ? resolve(n) : reject(n);
}, n*1000);
});
